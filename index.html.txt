  function renderTable() {
    const search = document.getElementById('searchInput').value.toLowerCase();
    let filtered = records.filter(r => {
      const matchSearch = r.student.toLowerCase().includes(search) || r.cls.toLowerCase().includes(search);
      if (!matchSearch) return false;
      if (currentFilter === 'all') return true;
      if (currentFilter === 'out') return r.status === 'out' && !isLate(r);
      if (currentFilter === 'in') return r.status === 'in';
      if (currentFilter === 'late') return isLate(r);
      return true;
    });

    const tbody = document.getElementById('trackingBody');
    const empty = document.getElementById('emptyState');

    if (filtered.length === 0) {
      tbody.innerHTML = '';
      empty.style.display = 'block';
      return;
    }

    empty.style.display = 'none';
    tbody.innerHTML = filtered.map((r, i) => {
      const elapsed = getElapsed(r.exitTime, r.returnTime);
      const late = isLate(r);
      let statusHtml = '';
      let statusClass = '';

      if (r.status === 'in') {
        statusHtml = '<span class="status-badge status-in">✅ عاد للفصل</span>';
        statusClass = '';
      } else if (late) {
        statusHtml = '<span class="status-badge status-late">⚠️ تأخر</span>';
      } else {
        statusHtml = '<span class="status-badge status-out">🔴 خارج</span>';
      }

      const timerClass = r.status === 'in' ? 'ok' : (late ? 'danger' : 'warn');
      const timerText = r.status === 'in'
        ? `<span class="timer ok">${elapsed}</span>`
        : `<span class="timer ${timerClass}">${elapsed}</span>`;

      const actionHtml = r.status === 'out'
        ? `<button class="return-btn" onclick="openReturnModal(${r.id})">🏠 عاد للفصل</button>`
        : `<span class="no-btn">—</span>`;

      return `<tr>
        <td style="color:var(--text-muted);font-weight:700">${i+1}</td>
        <td style="font-weight:700">${r.student}</td>
        <td style="font-weight:600;color:var(--primary)">${r.cls}</td>
        <td>${r.reason}${r.subject ? ' <span style="color:var(--text-muted);font-size:12px">(' + r.subject + ')</span>' : ''}</td>

  function isLate(r) {
    if (r.status === 'in') return false;
    const elapsed = (new Date() - r.exitTime) / 1000 / 60;
    return elapsed > r.duration;
  }

  function getElapsed(exit, ret) {
    const end = ret || new Date();
    const diff = Math.floor((end - exit) / 1000);
    const m = Math.floor(diff / 60);
    const s = diff % 60;
    return `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  }

  function formatTime(d) {
    return d.toLocaleTimeString('ar-SA', {hour:'2-digit', minute:'2-digit'});
  }

    document.getElementById('sumLate').textContent = late;
    document.getElementById('hdrOut').textContent = out + late;
    document.getElementById('hdrTotal').textContent = records.length;
  }
